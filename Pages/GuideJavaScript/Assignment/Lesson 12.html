  • Создайте функцию, принимающую двумерный массив и выводящую его в теге <table> на страницу, чтобы каждое число было в своей ячейке.

    • Код

    function OutputArrayInTable(InputArr) {
    if (typeof(InputArr) == "object") {
    document.write("<table border=1");
    for (i = 0; i < InputArr.length; i++) {
    document.write("<tr>")
    for (j = 0; j < InputArr[i].length; j++) {
    document.write("<td>" + InputArr[i][j] + "</td>");
    }
    document.write("</tr>");
    }
    document.write("</table>");
    } else
    document.write("Ошибка в форме передаваемых данных функции OutputArrayInTable");
    }
