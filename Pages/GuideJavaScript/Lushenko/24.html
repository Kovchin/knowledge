Таймеры и интервалы

В данном уроке рассматриваются таймеры и интервалы

Таймер позволяющий запускать действие раз в указанный период времени setInterval

Инициализация таймера с присвоением ему имени timer (что бы потом можно ыло к нему обратиться)

var timer = setInterval(Название функции , интервал запуска в миллисекундах);

Сброс действия по таймеру

clearInterval(timer);

Однократное, отложенное исполнение функции setTimeout

Данная функция однократно исполняет указанную функцию с задержкой исполнения

var timer = setTimeout(Название функции , интервал запуска в миллисекундах);

Отличается от предыдущей тем что первая однозначно запустится в указанный промежуток времени без учета времени исполнения самой функции

Эта же запускается однократно, ее можно внести в бесконечный цикл, но даже в этом случае таймер будет отсчитываться от момента вызова, то есть будет учтено время исполнения вызывающей функции.

Пример зацикливания команды setTimtout

var sdvig3 = 0; //-позиция 3-го блока
var test3 = document.getElementById('test3');//-блок который будет перемещаться
var timer3 = setTimeout(move3, 30);//-запускается таймер для однократного перемещения блока test3
function move3(){
//Функция движения блока вызываемая с помощью команды setTimeout
test3.style.marginLeft = sdvig3 + 'px';
if (sdvig3 >=100)
sdvig3=0;
else
sdvig3++;
setTimeout(move3, 30);
}
var MaxTimer4 = 10;//задается максимальное значение таймера обратного отчета.
var timer4 = setInterval(move4, 1000);//-первый запуск таймера обратного отчета
function move4(){
MaxTimer4 --;
if (MaxTimer4 < 0){
test4.innerHTML = 'Таймер остановлен!';
clearInterval(timer4);
}
else
test4.innerHTML = MaxTimer4;
}
