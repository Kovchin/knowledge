В этом уроке рассматривается работа с локальной базой данных страницы LocalStorage

LocalStorage - локальная база данных страницы. Она позволяет запоминать данные на стороне клиента для того что бы использовать их при следующих запусках браузера

localStorage

Это массив (ключ, значение)

  • localStorage.setItem('ключ' , 'значение') - Записать в localStorage
  • localStorage.getItem('ключ' , 'значение') - Прочитать из localStorage

Код

window.onload = function(){
document.getElementsByClassName('changedBlock')[0].style.background = localStorage.getItem('bgcolor');
}

document.getElementById('red').onclick = function(){
document.getElementsByClassName('changedBlock')[0].style.background = 'red';
localStorage.setItem('bgcolor', 'red');
}

document.getElementById('blue').onclick = function(){
document.getElementsByClassName('changedBlock')[0].style.background = 'blue';
localStorage.setItem('bgcolor', 'blue');
}

document.getElementById('orange').onclick = function(){
document.getElementsByClassName('changedBlock')[0].style.background = 'orange';
localStorage.setItem('bgcolor', 'orange');
}
